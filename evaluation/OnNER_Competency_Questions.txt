
1. Which "chemicals" are mentioned in conjunction with "permeability"? In which publications and paragraphs?

SELECT ?chemical ?paragraph ?publication WHERE {
    ?publication onner:containsDocumentPart ?paragraphID .
    
    ?paragraphID rdf:type onner:Paragraph ;
                 onner:paragraphText ?paragraph ;
                 onner:directlyContainsLabeledTerm ?ne1 , ?ne2 .
    
    ?ne1 onner:labeledTermText 'permeability'^^xsd:string .
    
    ?ne2 onner:labeledTermText ?chemical ;
         onner:hasLabeledTermStatus ?ne2Status .
    ?ne2Status onner:hasLabeledTermLabel ?label2 .
    ?label2 onner:labelText 'CHEMICAL'^^xsd:string . 
}

7. Give me a list of terms labeled as "property" from a particular publication (or all publications by a particular author) with their frequencies, starting with the highest frequency.

SELECT ?labeledTermText (COUNT(?labeledTermText) AS ?labeledTermTextFrequency)
WHERE {
    ?publicationID rdf:type onner:ScholarlyPublication ;
                   onner:publicationTitle 'How the shape of fillers affects the barrier properties of polymer/non-porous particles nanocomposites: A review'^^xsd:string .
    
    ?labeledTermID rdf:type onner:LabeledTerm ;
                   onner:labeledTermText ?labeledTermText ;
                   onner:hasLabeledTermStatus ?labeledTermStatus .
    
    ?labeledTermStatus onner:hasLabeledTermLabel ?labelID .
    
    ?labelID rdf:type onner:Label ;
             onner:labelText 'PROPERTY'^^xsd:string .
}
GROUP BY ?labeledTermText
ORDER BY DESC(?labeledTermTextFrequency)



8. What are the labeled terms that contain "cellulose" as a part of that term.

SELECT ?labeledTermID ?paragraphID ?labeledTermText ?paragraphText
WHERE {
    ?paragraphID rdf:type onner:Paragraph ;
                 onner:paragraphText ?paragraphText ;
                 onner:directlyContainsLabeledTerm ?labeledTermID .
    
	?labeledTermID rdf:type onner:LabeledTerm ;
                   onner:labeledTermText ?labeledTermText ;
                   onner:hasLabeledTermStatus ?labeledTermStatus .
    
    ?labeledTermStatus onner:hasLabeledTermLabel ?labelID .
    
    ?labelID rdf:type onner:Label ;
             onner:labelText ?labelText .
    
    FILTER(CONTAINS(?labeledTermText, 'cellulose') && ?labeledTermText != 'cellulose')
}

10. Have any of the labeled terms been classified by NER system with two different classification labels?

SELECT DISTINCT ?ne ?label1 ?label2
WHERE {
  ?ne rdf:type onner:LabeledTerm ;
      onner:hasLabeledTermStatus ?neStatus1, ?neStatus2 .
    
  ?neStatus1 onner:hasLabeledTermLabel ?label1 .
    
  ?neStatus2 onner:hasLabeledTermLabel ?label2 .
    
  FILTER(?label1 != ?label2)
}



11. Retrieve all paragraphs from publications where "cellulose" is labeled as a "material" alongside any other labeled term containing "cellulose".

SELECT ?paragraphID ?labeledTermID2 ?labeledTermText2 ?labelText2 ?paragraphText
WHERE { 
    ?paragraphID rdf:type onner:Paragraph ;
                 onner:paragraphText ?paragraphText ;
                 onner:directlyContainsLabeledTerm ?labeledTermID1 ;
                 onner:directlyContainsLabeledTerm ?labeledTermID2 .
    
	?labeledTermID1 rdf:type onner:LabeledTerm ;
                    onner:labeledTermText 'cellulose'^^xsd:string ;
                    onner:hasLabeledTermStatus ?labeledTermStatus1 .
    
    ?labeledTermID2 rdf:type onner:LabeledTerm ;
                    onner:labeledTermText ?labeledTermText2 .
                                                       
    ?labeledTermStatus1 onner:hasLabeledTermLabel ?labelID1 .
    
    ?labeledTermStatus2 onner:hasLabeledTermLabel ?labelID2 .
    
    ?labelID1 rdf:type onner:Label ;
              onner:labelText 'MATERIAL'^^xsd:string .
    
    ?labelID2 rdf:type onner:Label ;
              onner:labelText ?labelText2 .
    
    FILTER(CONTAINS(?labeledTermText2, 'cellulose') && ?labeledTermText2 != 'cellulose')
}
